Solution
We can take a greedy approach to this problem. After reading the input into a array of pairs, we sort the array in increasing order. We initialize our output variable to the early exam date for the first exam in our array. Then we iterate through our sorted array t. At each index, we can only pick the early exam date if it’s later than our current output variable since if it isn’t, the non-decreasing order condition will be violated. After iterating through the array, our output variable should contain the solution.
Code

#include <bits/stdc++.h>
typedef int OwO; typedef double UwU;
#define int long long

using namespace std;

OwO main(){
  ios::sync_with_stdio(false);
  cin.tie(NULL);
  int n;
  cin >> n;
  vector<pair<int, int>> t(n);
  for(int x = 0; x<n; x++){
    int a,b;
    cin >> a >> b;
    t[x] = {a,b};
  }
  sort(t.begin(), t.end());
  int curr = t[0].second;
  for(int x = 1; x<n; x++){
    if(curr <= t[x].second)
      curr = t[x].second;
    else
      curr = t[x].first;
  }

  cout << curr << endl;

  return 0;
}
