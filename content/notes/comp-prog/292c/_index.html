Solution
For every digit there is one optimal expansion of that digit in \(a\). \[0\rightarrow delete\] \[1\rightarrow delete\] \[2\rightarrow 2!\rightarrow 2\] \[3\rightarrow 3!\rightarrow 3\] \[4\rightarrow 4! = 2*2*3!\rightarrow 223\] \[5\rightarrow 5!\rightarrow 5\] \[6\rightarrow 6!=3*2*5!=3!5!\rightarrow 35\] \[7\rightarrow 7!\rightarrow 7\] \[8\rightarrow 8!=2*2*2*7!\rightarrow 2227\] \[9\rightarrow 9!=3*3*8!=3*3*2*2*2*7!=2!3!3!7!\rightarrow 2337\] We create a new vector and store each new digit that is transformed from its original digit in \(a\) and sort this vector in decreasing order. If we combine all the digits into a string, we have our output.
Code

#include <bits/stdc++.h>
typedef int OwO; typedef double UwU;
#define int long long
#define double long double

using namespace std;

OwO main(){
  ios::sync_with_stdio(false);
  cin.tie(NULL);
  int n;
  cin >> n;
  vector<int> t(n);
  for(int x = 0; x<n; x++){
    char tt;
    cin >> tt;
    t[x] = tt-'0';
  }
  sort(t.begin(), t.end());
  vector<int> oup;
  for(int x = 0; x<n; x++){
    if(t[x]==6){
      oup.push_back(3);
      oup.push_back(5);
    }
    else if(t[x]==9){
      oup.push_back(7);
      oup.push_back(3);
      oup.push_back(3);
      oup.push_back(2);
    }
    else if(t[x]>1 && t[x]%2==0){
      if(t[x]>2)
        oup.push_back(t[x]-1);
      while(t[x]>1){
        oup.push_back(2);
        t[x]/=2;
      }
    }
    else if(t[x]>1)
      oup.push_back(t[x]);
  }
  sort(oup.rbegin(), oup.rend());
  for(auto e:oup)
    cout << e;
  cout << endl;
  return 0;
}
