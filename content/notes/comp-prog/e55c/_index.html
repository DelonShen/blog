Solution
We can take a greedy approach to this problem. First we read in everyone and sort them in decreasing order by their skill. Then we create an array, oup, where oup[s-1][j] is the maximum skill for event s-1 given j people. We have another array trk where trk[j] is the sum of all the skills if we only use teams of size \(j\). To fill these arrays up we iterate through our sorted people and put them on the team only if the resulting total skill for that event is greater than 0. Once we iterate through all the people, we sort trk in decreasing order and output the first element in trk.
Code

#include <bits/stdc++.h>
typedef int OwO; typedef double UwU;
#define int long long
#define double long double

using namespace std;

OwO main(){
  ios::sync_with_stdio(false);
  cin.tie(NULL);

  int n,m;
  cin >> n >> m;

  vector<pair<int, int>> init(n);
  for(int x = 0; x<n; x++){
    int r,s;
    cin >> s >> r;
    init[x]={r,s};
  }
  sort(init.rbegin(), init.rend());

  vector<vector<int>> oup(m+1, vector<int>{0});
  vector<int> trk(n+10);

  for(int x = 0; x<n; x++){
    int s=init[x].second, r=init[x].first;
    int quant = oup[s-1][oup[s-1].size()-1]+r;
    if(quant>0){
      trk[oup[s-1].size()]+=quant;
      oup[s-1].push_back(quant);
    }
  }
  sort(trk.rbegin(), trk.rend());
  cout << trk[0] << endl;

  return 0;
}
