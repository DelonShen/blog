Solution
For this problem we can brute-force the solution. For every ’AB’ substring we find, we search past this substring to try to find a ’BA’ substring and for every ’BA’ substring we find, we search past this substring to find an ’AB’ substring. If both ’AB’ and ’BA’ are found, we output YES and terminate the program. There’s a more optimal solution described in the editorial.
Code

#include <bits/stdc++.h>
typedef int OwO; typedef double UwU;
#define int long long
#define double long double

using namespace std;

OwO main(){
  ios::sync_with_stdio(false);
  cin.tie(NULL);
  string s;
  cin >> s;
  if(s.size()<4){
    cout << "NO" << endl;
    return 0;
  }
  for(int x = 0; x<s.size()-3; x++){
    if(s[x]=='A' && s[x+1]=='B'){
      for(int y=x+2; y<s.size()-1; y++){
        if(s[y]=='B' && s[y+1]=='A'){
          cout << "YES" << endl;
          return 0;
        }
      }
    }
    else if(s[x]=='B' && s[x+1]=='A'){
      for(int y=x+2; y<s.size()-1; y++){
        if(s[y]=='A' && s[y+1]=='B'){
          cout << "YES" << endl;
          return 0;
        }
      }
    }
  }
  cout << "NO" << endl;
  return 0;
}
